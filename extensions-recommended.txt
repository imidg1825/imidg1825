# Рекомендованные расширения VS Code для будущего развития
# Автор: Иван (imidg1825)

# --- Python: форматирование и линтинг ---

ms-python.black-formatter
    # Автоформатирование Python-кода через Black.
    # Полезно для единообразного стиля и чистых коммитов.

charliermarsh.ruff
    # Ruff — сверхбыстрый линтер/форматтер.
    # Стиль кода, поиск ошибок, дублирования, лишних импортов.

# --- YAML / CI/CD / DevOps ---

redhat.vscode-yaml
    # Полная поддержка YAML: автодополнение, диагностика.
    # Нужно для docker-compose, GitHub Actions, Kubernetes и т.п.

github.vscode-github-actions
    # Поддержка GitHub Actions workflow-файлов.
    # Автодополнение, подсветка ошибок, структура.

# --- Улучшение Markdown / документации ---

yzhang.markdown-all-in-one
    # Прокачка Markdown: автонумерация, предпросмотр, горячие клавиши.
    # Полезно для README, документации, отчётов.

# --- SSH / удалённая разработка (когда появится сервер) ---

ms-vscode-remote.remote-ssh
    # Позволяет открывать проекты по SSH прямо в VS Code.
    # Нужно, если будешь подключаться к удалённому Linux-серверу.

# --- Полезные, но НЕ обязательные ---

formulahendry.code-runner
    # Быстрый запуск кода (Python, JS и других языков) без настройки конфигурации.

streetsidesoftware.code-spell-checker
    # Проверка орфографии в комментариях и Markdown.
    # Полезно для красивых README.
